#include <iostream>
#include <algorithm>
using namespace std;

int N;
int num[1000001];

int fixPoint() {
	int start = 0;
	int end = N - 1;

	while (start <= end) {
		int mid = (start + end) / 2;

		if (num[mid] > mid)
			end = mid - 1;
		else if (num[mid] < mid)
			start = mid + 1;
		else if (num[mid] == mid)
			return mid;
	}
	return -1;
}

int main() {
	cin >> N;
	for (int i = 0; i < N; i++)
		scanf("%d", &num[i]);

	sort(num, num + N);

	cout << fixPoint();
}
