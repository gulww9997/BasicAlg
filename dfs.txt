#include <iostream>
#include <vector>

using namespace std;

bool visited[9];
vector<int> graph[9];

void dfs(int x) {
	visited[x] = true;
	cout << x << " ";
	for (int i = 0; i < graph[x].size(); i++) {
		int next = graph[x][i];
		if (!visited[next])
			dfs(next);
	}
}

int main() {
	/*
	그래프 연결 내용 생략
	
	*/
	dfs(1);
}
