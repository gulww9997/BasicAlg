#include <iostream>
#include <algorithm>
using namespace std;

int N, M;
int parts[100000];
int user[100000];

int binarySearch(int low, int high, int target) {
	while (low <= high) {
		int mid = (low + high) / 2;

		if (parts[mid] < target)
			low = mid + 1;
		else if (parts[mid] > target)
			high = mid - 1;
		else if (parts[mid] == target)
			return 1;
	}
	return 0;
}


int main() {
	cin >> N;
	for (int i = 0; i < N; i++)
		scanf("%d", &parts[i]);

	cin >> M;
	for (int i = 0; i < M; i++)
		scanf("%d", &user[i]);

	sort(parts, parts + N);

	for (int i = 0; i < M; i++) {
		if (binarySearch(0, N - 1, user[i]) == 1)
			printf("yes ");
		else
			printf("no ");
	}
}
