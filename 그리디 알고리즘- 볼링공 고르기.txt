#include <iostream>
#include <map>
using namespace std;

int weight[1000];

int main() {
	/*
	* 시간복잡도 O(N^2)으로 비효율적

	int N, M, count = 0;
	cin >> N >> M;
	for (int i = 0; i < N; i++)
		cin >> weight[i];

	for (int i = 0; i < N - 1; i++) {
		for (int j = i + 1; j < N; j++) {
			if (weight[i] != weight[j])
				count++;
		}
	}

	cout << count;
	*/
	
	int N, M, index, count = 0;
	cin >> N >> M;
	map<int, int> m;

	for (int i = 0; i < N; i++) {
		cin >> index;
		m[index] += 1;
	}

	for (int i = 1; i <= M; i++) {
		N -= m[i];
		count += m[i] * N;
	}

	cout << count;
}
