#include <iostream>
#include <vector>
#include <queue>

using namespace std;

bool visited[9];
vector<int> graph[9];

void bfs(int x) {
	queue<int> q;
	q.push(x);
	visited[x] = true;

	while(!q.empty()){
		int now = q.front();
		q.pop();
		cout << now << " ";
		for (int i = 0; i < graph[x].size(); i++) {
			int next = graph[x][i];
			if (!visited[next]) {
				q.push(next);
				visited[next] = true;
			}
		}
	}
}

int main() {
	/*
	그래프 연결 내용 생략
	
	*/
	bfs(1);
}
