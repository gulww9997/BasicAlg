#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	string s;
	cin >> s;
	vector<char> v;
	int sum = 0;
	for (int i = 0; i < s.length(); i++) {
		if (s[i] >= 'A' && s[i] <= 'z')
			v.push_back(s[i]);
		else
			sum += s[i] - '0';
	}
	sort(v.begin(), v.end());
	for (auto i = v.begin(); i != v.end(); i++)
		cout << *i;
	cout << sum;
}
