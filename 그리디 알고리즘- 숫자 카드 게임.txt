#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int N, M, max = 0;
	cin >> N >> M;
	int* arr = new int[M];

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			cin >> arr[j];
		}
		sort(arr, arr + M);
		if (max < arr[0])
			max = arr[0];
	}

	cout << max;
}
