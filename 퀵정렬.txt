#include <iostream>
using namespace std;

void quickSort1(int* data, int start, int end) { // 오름차순

	if (start >= end) // 원소가 1개인 경우
		return;

	int pivot = start;
	int i = start + 1;
	int j = end;
	int temp;

	while (i <= j){  // 엇갈릴 때까지 반복
		while (data[i] <= data[pivot] && i <= end) // pivot보다 큰 값을 만날때 까지 오른쪽으로
			i++;
		while (data[j] >= data[pivot] && j > start) // pivot보다 작은 값을 만날때 까지 왼쪽으로
			j--;

		if (i > j) { // 엇갈렸다면
			temp = data[j];
			data[j] = data[pivot];
			data[pivot] = temp;
		}
		else { // 엇갈리지 않았다면
			temp = data[i];
			data[i] = data[j];
			data[j] = temp;
		}
	}

	quickSort1(data, start, j - 1);
	quickSort1(data, j + 1, end);
}

int main() {
	int number = 10;
	int array[10] = { 1,10,5,8,7,6,4,3,2,9 };

	quickSort1(array, 0, number - 1);
	for (int i = 0; i < number; i++)
		cout << array[i] << " ";

	cout << endl;

	quickSort2(array, 0, number - 1);
	for (int i = 0; i < number; i++)
		cout << array[i] << " ";
}
