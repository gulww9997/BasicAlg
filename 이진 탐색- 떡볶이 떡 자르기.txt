#include <iostream>
#include <algorithm>
using namespace std;

int N, M;
int sum;
int result;
int len[1000000];

void maxlen() {
	int low = 0;
	int high = len[N - 1];

	while (low <= high) {
		int cut = (low + high) / 2; // 절단 길이 조절
		sum = 0;

		for (int i = 0; i < N; i++) // 떡의 양 계산
			if (len[i] > cut)
				sum += len[i] - cut;

		if (sum >= M) { // 떡의 양이 충분한 경우
			result = cut; 
			low = cut + 1;
		}
		else if (sum < M) { // 떡의 양이 부족한 경우
			high = cut - 1;
		}
	}
}

int main() {
	cin >> N >> M;
	for (int i = 0; i < N; i++)
		scanf("%d", &len[i]);

	sort(len, len + N); // 이진 탐색을 위한 정렬

	maxlen();
	cout << result;
}
