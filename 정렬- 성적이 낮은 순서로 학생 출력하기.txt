#include <iostream>
#include <vector>
using namespace std;

vector<pair<int, string>> student;


bool compare(pair<int,string> a, pair<int,string> b) {
	return a.first < b.first;
}

int main() {
	int N;
	cin >> N;
	
	for (int i = 0; i < N; i++) {
		int score;
		string name;
		cin >> name >> score;
		student.push_back({ score,name });
	}

	sort(student.begin(), student.end(), compare);

	for (int i = 0; i < N; i++)
		cout << student[i].second << " ";
}
