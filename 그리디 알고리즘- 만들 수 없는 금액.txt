#include <iostream>
#include <algorithm>
using namespace std;

int coin[1000];

int main() {
	int N, target = 1, result = 0;
	cin >> N;
	for (int i = 0; i < N; i++)
		cin >> coin[i];

	sort(coin, coin + N);

	for (int i = 0; i < N; i++) {
		if (target >= coin[i])
			target += coin[i];
		else
			result = target;
	}

	cout << result;
}
