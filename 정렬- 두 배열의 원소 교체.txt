#include <iostream>
#include <algorithm>
using namespace std;

int A[100001];
int B[100001];

bool compare(int a, int b) {
	return a > b;
}

void swap(int& a, int& b) {
	int temp = a;
	a = b;
	b = temp;
}

int main() {
	int N, K;
	cin >> N >> K;

	for (int i = 0; i < N; i++)
		cin >> A[i];

	for (int i = 0; i < N; i++)
		cin >> B[i];

	sort(A, A + N);
	sort(B, B + N, compare);

	for (int i = 0; i < K; i++)
		if (A[i] < B[i])
			swap(A[i], B[i]);

	int sum = 0;

	for (int i = 0; i < N; i++)
		sum += A[i];

	cout << sum;
}
