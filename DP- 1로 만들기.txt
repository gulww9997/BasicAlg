#include <iostream>
#include <algorithm>
using namespace std;

int X;
int dp[30001];

int main() {
	cin >> X;

	for (int i = 2; i <= X + 1; i++) { // 1 더하는 것은 계산 횟수 1 증가의 의미
		dp[i] = dp[i - 1] + 1;

		if (i % 5 == 0) {
			dp[i] = min(dp[i], dp[i / 5] + 1);
		}

		if (i % 3 == 0) {
			dp[i] = min(dp[i], dp[i / 3] + 1);
		}

		if (i % 2 == 0) {
			dp[i] = min(dp[i], dp[i / 2] + 1);
		}
	}

	cout << dp[X];
}
