#include <iostream>
#include <queue>
using namespace std;

int N, M;
int map[201][201];

int dx[4] = { 1,-1,0,0 }; // 이동 할 수 있는 경우
int dy[4] = { 0,0,1,-1 };

int bfs(int x, int y) {
	queue<pair<int, int>> q;
	q.push({ x, y });
	while (!q.empty()) { // 큐가 빌 때까지 반복
		int x = q.front().first;
		int y = q.front().second;
		q.pop();
		for (int i = 0; i < 4; i++) {
			int nx = x + dx[i];
			int ny = y + dy[i];

			if (nx < 0 || nx > N || ny < 0 || ny > M) // 맵을 벗어난 경우
				continue;
			
			if (map[nx][ny] == 0) // 괴물이 있는 경우
				continue;

			if (map[nx][ny] == 1) {
				map[nx][ny] = map[x][y] + 1;
				q.push({ nx,ny });
			}
		}
	}
	return map[N - 1][M - 1];
}


int main() {
	cin >> N >> M;
	for (int i = 0; i < N; i++)
		for (int j = 0; j < M; j++)
			scanf("%1d", &map[i][j]);

	cout << bfs(0, 0);
}
