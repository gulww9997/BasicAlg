#include <iostream>
#include <string>
using namespace std;

struct locate {
	int x = 1;
	int y = 1;
};

int main() {
	int N;
	char s;
	struct locate now;
	cin >> N;
	cin.ignore();

	while (true) {
		s = cin.get();

		if (s == ' ')
			continue;
		if (s == '\n')
			break;
		if (s == 'L' && now.y != 1)
			now.y--;
		if (s == 'R' && now.y != N)
			now.y++;
		if (s == 'U' && now.x != 1)
			now.x--;
		if (s == 'D' && now.x != N)
			now.x++;
	}
	cout << now.x << " " << now.y;
}
