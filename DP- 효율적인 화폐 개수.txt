#include <iostream>
using namespace std;

int N, M;
int coin[101];
int dp[10001];

int main() {
	cin >> N >> M;
	for (int i = 0; i < N; i++)
		cin >> coin[i];

	dp[0] = 0;
	for (int i = 1; i <= M; i++)
		dp[i] = 10001;

	for (int i = 0; i < N; i++) {
		for (int j = 1; j <= M; j++) {

			if (j < coin[i])
				continue;

			if (dp[j - coin[i]] != 10001)
				dp[j] = min(dp[j], dp[j - coin[i]] + 1);
			
		}
	}

	if (dp[M] != 10001)
		cout << dp[M];
	else
		cout << -1;
}
