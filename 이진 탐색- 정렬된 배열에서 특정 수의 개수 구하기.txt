#include <iostream>
#include <algorithm>
using namespace std;

int N, x;
int firstIndex;
int lastIndex;
int num[1000001];

void first() {
	int start = 0;
	int end = N - 1;

	while (start <= end) {
		int mid = (start + end) / 2;

		if (num[mid] < x)
			start = mid + 1;
		else if (num[mid] > x)
			end = mid - 1;
		else if (num[mid] == x) {
			if (num[mid - 1] != x) {
				firstIndex = mid;
				break;
			}
			else
				end = mid - 1;
		}
	}
}

void last() {
	int start = 0;
	int end = N - 1;

	while (start <= end) {
		int mid = (start + end) / 2;

		if (num[mid] < x)
			start = mid + 1;
		else if (num[mid] > x)
			end = mid - 1;
		else if (num[mid] == x) {
			if (num[mid + 1] != x) {
				lastIndex = mid;
				break;
			}
			else
				start = mid + 1;
		}
	}
}

int main() {
	cin >> N >> x;

	for (int i = 0; i < N; i++)
		scanf("%d", &num[i]);

	sort(num, num + N);
	first();
	last();
	
	if (firstIndex == lastIndex)
		cout << "-1";
	else
		cout << lastIndex - firstIndex + 1;
}
